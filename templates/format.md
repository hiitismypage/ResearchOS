# Настройки форматирования

## Язык работы
language: ru
# Варианты: ru | en | de | fr | другой
# Влияет на: язык интерфейса скиллов, переносы слов, правила пунктуации

## Шрифт основного текста
font_family: Times New Roman
# Часто используются: Times New Roman | Arial | Calibri | Garamond
font_size: 14
# Обычно 12 или 14 для академических работ

## Заголовки
heading1_font: Times New Roman
heading1_size: 14
heading1_bold: true
heading1_uppercase: true
heading1_alignment: center

heading2_font: Times New Roman
heading2_size: 14
heading2_bold: true
heading2_uppercase: false
heading2_alignment: left

heading3_font: Times New Roman
heading3_size: 14
heading3_bold: false
heading3_italic: true
heading3_alignment: left

## Абзацы
line_spacing: 1.5
# Варианты: 1.0 | 1.5 | 2.0
first_line_indent: 1.25cm
paragraph_spacing_before: 0pt
paragraph_spacing_after: 0pt
body_alignment: justify
# Варианты: justify | left | center

## Поля страницы (в мм)
margin_top: 20
margin_bottom: 20
margin_left: 30
margin_right: 15
# Примечание: ГОСТ — левое 30мм, правое 15мм, верхнее и нижнее 20мм

## Страница
page_size: A4
page_numbers: bottom_center
# Варианты: bottom_center | bottom_right | top_right | none
page_numbers_start: 1
# Если нумерация начинается не с первой страницы, указать номер первой нумеруемой страницы

## Цитирование
citation_style: GOST
# Варианты: GOST | APA | MLA | Chicago | Vancouver | Harvard
citation_format: footnote
# Варианты: footnote | endnote | inline | bibliography_only

## Особые требования
custom_notes: |
  # Любые дополнительные требования от научного руководителя
  # Например: "Оглавление не нумеруется", "Введение начинается с нечётной страницы"
